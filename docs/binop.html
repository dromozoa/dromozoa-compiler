arithmetic/bitwise/concatenate binop

u = u1 {binop} u2

if "{opname}" == "arithmetic" then
  message = "attempt to perform arithmetic operation on a "
  convert = TONUMBER
elseif "{opname}" == "bitwise" then
  message = "attempt to perform bitwise operation on a "
  convert = TOINTEGER
elseif "{opname}" == "concatenate" then
  message = "attempt to perform concatenate on a "
  convert = TOSTRING
end

B v1
B v2
B v3
B v4
C v5
C v6
C v7
C v8
K v9 "__{binop:lower()}"
K v10 "{message}"
K v11 " value"

{convert} v5 u1
{convert} v6 u2

COND_IF v5
  MOVE v1 v6
COND_ELSE
  MOVE v1 v5
COND_END

COND_IF v1
  {binop} v2 v5 v6
COND_ELSE
  GETMETAFILED v3 u1 v9
  COND_IF v3
    ;
  COND_ELSE
    GETMETAFILED v3 u2 v9
  COND_END

  COND_IF v3
    CALL v3 u1 u2
    RESULT v2
  COND_ELSE
    COND_IF v5
      TYPENAME v4 u2
    COND_ELSE
      TYPENAME v4 u1
    COND_END
    CONCAT v7 v10 v4
    CONCAT v8 v7 v11
    ERROR v8 0
  COND_END
COND_END

MOVE u v2
