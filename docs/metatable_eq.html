  COND_IF c2
    TYPE c3 u2

    COND_IF c4
      GETMETAFIELD b2 u1 k1
      COND_IF b2
        ;
      COND_ELSE
        GETMETAFIELD b2 u2 k1
      COND_END
      COND_IF b2
        CALL b2 u1 u2
        RESULT c5
        COND_IF c5
          MOVE b1 true
        COND_END
      COND_END
    COND_END
  COND_END
COND_END
